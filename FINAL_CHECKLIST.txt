📚 USAGE EXAMPLE

import pandas as pd
import numpy as np

# Series example
s = pd.Series([1.0, np.nan, np.nan, np.nan, 5.0])

# Default behavior (unchanged)
s.interpolate(limit=1)
# [1.0, 2.0, NaN, NaN, 5.0]

# New skip behavior
s.interpolate(limit=1, limit_behavior="skip")
# [1.0, NaN, NaN, NaN, 5.0]

# DataFrame example
df = pd.DataFrame({
    "A": [1.0, np.nan, np.nan, np.nan, 5.0],
    "B": [2.0, np.nan, 6.0, np.nan, np.nan],
})

df.interpolate(limit=1, limit_behavior="skip", axis=0)
# Applies skip behavior independently to each column
